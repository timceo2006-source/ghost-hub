local Window = WindUI:CreateWindow({
    Title = "Ghost Hub",
    Icon = "door-open",
    Author = "TIM",
    Folder = "MyGhostHub",
    Size = UDim2.fromOffset(580, 460),
    Theme = "Dark",
})

local MainTab = Window:Tab({ Title = "Tab Title", Icon = "bird" })

-- [1] ประกาศตัวแปรควบคุม
local autoFarmActive = false
local loopTask = nil

-- [2] ฟังก์ชันหลัก (รวม Logic ทั้งหมดไว้ที่นี่)
local function runFarmLogic()
    local Players = game:GetService("Players")
    local Player = Players.LocalPlayer
    local lockedTarget = nil

    -- ประกาศฟังก์ชันย่อยไว้ข้างในเพื่อให้ลูปมองเห็น
    local function getQuestInfo()
        local level = Player.PlayerData.Experience.Level.Value 
        if level >= 25 then return "Level 25", "Bandit Boss"
        elseif level >= 10 then return "Level 10", "Bandit"
        else return "Level 1", "Thief" end
    end

    local function acceptQuest(questName)
        local NpcWorkspace = workspace:FindFirstChild("Npc_Workspace")
        local QuestGiver = NpcWorkspace and NpcWorkspace:FindFirstChild("QuestGivers") and NpcWorkspace.QuestGivers:FindFirstChild("1")
        local questFunc = Player:FindFirstChild("PlayerGui") and Player.PlayerGui:FindFirstChild("QuestGui") and Player.PlayerGui.QuestGui:FindFirstChild("QuestFunction")
        if questFunc and QuestGiver then pcall(function() questFunc:InvokeServer(QuestGiver, questName) end) end
    end

    loopTask = task.spawn(function()
        while autoFarmActive do
            local Character = Player.Character
            if not Character or not Character:FindFirstChild("Humanoid") or Character.Humanoid.Health <= 0 then
                task.wait(2); continue
            end

            local tool = Character:FindFirstChild("Combat") or Player.Backpack:FindFirstChild("Combat")
            if not tool then task.wait(10); if Character:FindFirstChild("Humanoid") then Character.Humanoid.Health = 0 end; continue end
            
            local TargetFolder = workspace:FindFirstChild("NPC Zones") and workspace["NPC Zones"]:FindFirstChild("Starter Island") and workspace["NPC Zones"]["Starter Island"]:FindFirstChild("NPCS")
            
            if TargetFolder then
                local currentLevel = Player.PlayerData.Experience.Level.Value
                local _, targetName = getQuestInfo()
                if Player.Quest.GiverName.Value == "" then acceptQuest(select(1, getQuestInfo())) end

                -- ระบบหาเป้าหมายเดิมหรือหาใหม่
                if not (lockedTarget and lockedTarget:FindFirstChild("Humanoid") and lockedTarget.Humanoid.Health > 0) then
                    lockedTarget = nil
                    for _, obj in pairs(TargetFolder:GetChildren()) do
                        local isTarget = string.match(obj.Name, "^" .. targetName)
                        local isBoss = string.find(string.lower(obj.Name), "boss") ~= nil
                        if isTarget and obj:FindFirstChild("Humanoid") and obj.Humanoid.Health > 0 then
                            if (currentLevel < 25 and not isBoss) or (currentLevel >= 25 and isBoss) then
                                lockedTarget = obj; break
                            end
                        end
                    end
                end

                -- โจมตี
                if lockedTarget and lockedTarget:FindFirstChild("HumanoidRootPart") then
                    Character.Humanoid:ChangeState(Enum.HumanoidStateType.Physics)
                    Character.HumanoidRootPart.CFrame = CFrame.lookAt(lockedTarget.HumanoidRootPart.Position + Vector3.new(0, 7, 0), lockedTarget.HumanoidRootPart.Position)
                    Character.HumanoidRootPart.AssemblyLinearVelocity = Vector3.zero
                    if not Character:FindFirstChild("Combat") then Character.Humanoid:EquipTool(tool) end
                    tool:Activate()
                else
                    Character.Humanoid:ChangeState(Enum.HumanoidStateType.GettingUp)
                end
            end
            task.wait(0.2)
        end
    end)
end

-- [3] Callback ของ Toggle
MainTab:Toggle({
    Title = "AutoFarm",
    Desc = "AutoFarmLevel",
    Value = false,
    Callback = function(state)
        autoFarmActive = state
        if state then
            runFarmLogic()
        else
            if loopTask then task.cancel(loopTask) end
        end
    end
})
